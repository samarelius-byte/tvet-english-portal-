// api/chat.js
import fetch from 'node-fetch';

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  const { message, cefrLevel, tvetField } = req.body;

  // Validate input
  if (!message || !cefrLevel) {
    return res.status(400).json({ error: 'Missing required fields' });
  }

  // Your DeepSeek API Key (only visible here, not in browser)
  const API_KEY = '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'; // üîê Replace with your key
  const API_URL = 'https://api.deepseek.com/v1/chat/completions';

  try {
    // Construct system message based on level & field
    const systemContent = `You are an AI English tutor for Malaysian TVET students at ${cefrLevel} level (CEFR).
    Student's field: ${tvetField || 'General TVET'}.
    Respond in the same language as the user (English or Bahasa Malaysia).
    Use real-world examples from their field when possible.
    Be encouraging, clear, and pedagogical.
    Use Markdown: **bold**, *italic*, lists.`;

    const response = await fetch(API_URL, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${API_KEY}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'deepseek-chat',
        messages: [
          { role: 'system', content: systemContent },
          { role: 'user', content: message }
        ],
        max_tokens: 512,
        temperature: 0.7
      })
    });

    const data = await response.json();

    if (data.choices && data.choices[0]?.message?.content) {
      return res.status(200).json({ reply: data.choices[0].message.content });
    } else {
      return res.status(500).json({ error: 'Invalid response from AI' });
    }
  } catch (error) {
    console.error('AI API Error:', error);
    return res.status(500).json({ error: 'Failed to connect to AI' });
  }
}
